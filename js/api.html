JSON.parse(s) : []; } } async function registerUser(name, email, password) { try { const res = await fetch(API.POST_USER, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ Name: name, Email: email, Password: password, Role: 'Viewer' }) }); return await res.json(); } catch (err) { console.error('Failed to register user:', err); return null; } } // =========================================== // SUBMISSIONS // =========================================== async function fetchSubmissions() { try { const res = await fetch(API.GET_SUBMISSIONS); const data = await res.json(); return data.map(s => ({ formId: s.FormID, lastName: s.LastName, firstName: s.FirstName, mi: s.MI, office: s.Office, telephone: s.Telephone, date: s.DateSubmitted, purpose: s.Purpose, description: s.Description, period: s.Period, fromDate: s.FromDate, toDate: s.ToDate, status: s.Status || 'Pending', actionedBy: s.ActionedBy, actionedByRole: s.ActionedByRole, actionedAt: s.ActionedAt, initialConduct: s.InitialConduct, regularConduct: s.RegularConduct, asRequested: s.AsRequested })); } catch (err) { console.error('Failed to fetch submissions:', err); const s = localStorage.getItem('snarfFormSubmissions'); return s ? JSON.parse(s) : []; } } async function updateSubmission(formId, updates) { try { const res = await fetch(API.UPDATE_SUBMISSION, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ FormID: formId, ...updates }) }); return await res.json(); } catch (err) { console.error('Failed to update submission:', err); return null; } }
